       export const ARROW_VECTORS = {
         ArrowRight: { dx: 1, dy: 0 },
         ArrowLeft: { dx: -1, dy: 0 },
         ArrowDown: { dx: 0, dy: 1 },
         ArrowUp: { dx: 0, dy: -1 },
       }

       export function findNextNode(currentId, direction, nodes, layout) {
         if (!nodes || nodes.length === 0) return currentId
         if (!layout || !layout[currentId]) return currentId
         const vec = ARROW_VECTORS[direction]
         if (!vec) return currentId
         const curr = layout[currentId]
         const candidates = nodes.filter((n) => {
           if (n.id === currentId) return false
           const p = layout[n.id]
           if (!p) return false
           return (p.x - curr.x) * vec.dx + (p.y - curr.y) * vec.dy > 0
         })
         if (candidates.length === 0) {
           // Wrap: pick the farthest node in the opposite half-plane
           const others = nodes.filter((n) => n.id !== currentId && layout[n.id])
           if (others.length === 0) return currentId
           return others.reduce((best, n) => {
             const p = layout[n.id]
             const bp = layout[best.id]
             const dotN = (p.x - curr.x) * vec.dx + (p.y - curr.y) * vec.dy
             const dotB = (bp.x - curr.x) * vec.dx + (bp.y - curr.y) * vec.dy
             return dotN < dotB ? n : best
           }, others[0]).id
         }
         return candidates.reduce((best, n) => {
           const p = layout[n.id]
           const bp = layout[best.id]
           const d = Math.hypot(p.x - curr.x, p.y - curr.y)
           const bd = Math.hypot(bp.x - curr.x, bp.y - curr.y)
           return d < bd ? n : best
         }, candidates[0]).id
       }
       ```

            cx={pos.x}
            cy={pos.y}
            r={r + 4}
            fill="none"
            stroke="var(--color-brand)"
            strokeWidth="2"
            className="pointer-events-none"
            aria-hidden="true"
          />
        )}
        {/* Slice 3 pulse on biggest node — motion-safe only, stops on first interaction */}
        {node.id === biggestNodeId && !hasInteracted && (
          <circle
            cx={pos.x}
            cy={pos.y}
            r={r + 2}
            fill="none"
            stroke={SKILL_CATEGORY_COLORS[node.category]}
            strokeOpacity="0.5"
            strokeWidth="2"
            className="motion-safe:animate-pulse2"
            aria-hidden="true"
          />
        )}
      </g>
      ```
      `biggestNodeId` = the id of the node with max count (compute once via `useMemo`: `const biggestNodeId = useMemo(() => nodes.length === 0 ? null : nodes.reduce((m, n) => (!m || n.count > m.count ? n : m), null)?.id, [nodes])`).

      The `motion-safe:animate-pulse2` class is gated by `hasInteracted` — once any user action fires (click, focus, hover, key), `hasInteracted = true` and the pulse disappears from the DOM. Reduced-motion users never see the pulse (the motion-safe Tailwind prefix suppresses animations under reduced-motion globally).

      **Hint pill** (sibling of `<svg>` inside the root wrapper, below the SVG):
      ```
      {prefersReducedMotion && !hasInteracted && (
        <p
          className="text-center mt-4 text-xs font-mono text-hintPill-text bg-hintPill-bg rounded-full px-3 py-2 inline-block"
          aria-hidden="true"
        >
          {t.game.hintPill}
        </p>
      )}
      ```
      (Per 15-UI-SPEC.md §Hint pill lines 374-386 — show ONLY under reduced-motion; otherwise the pulse handles the affordance.)

      ```

        {announcement}
      </div>
      ```

      **Selection-change effect** (sets the announcement string):
      ```
      useEffect(() => {
        if (selectedSkillId === null) {
          // Suppress on mount (initial null); only emit "cleared" after a prior selection.
          // Track previousSelectedRef.current; if it was non-null and current is null → emit selectionCleared.
          // Simplest: gate on hasInteracted to avoid the mount-clear announcement.
          if (hasInteracted) setAnnouncement(t.game.selectionCleared)
          return
        }
        const node = nodes.find((n) => n.id === selectedSkillId)
        if (!node) return
        const noun = node.count === 1 ? t.game.nodeJobSingular : t.game.nodeJobPlural
        // Build the template (EN/ES use the same shape with different copy):
        const text = t.game.skillSelected
          .replace('{name}', node.label)
          .replace('{n}', String(node.count))
          .replace('{s}', node.count === 1 ? '' : 's')
        // Note: ES strings use "experiencia{s}" → "experiencias" with {s}='s'; English "experience{s}" → "experiences".
        // Same {s} works for both per spec template.
        setAnnouncement(text)
      }, [selectedSkillId, hasInteracted, lang, t, nodes])
      ```
