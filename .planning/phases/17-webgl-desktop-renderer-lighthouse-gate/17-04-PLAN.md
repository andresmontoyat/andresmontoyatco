    Pointer-pick useEffect (canvas listener):
    ```
    useEffect(() => {
      const canvas = canvasRef.current
      if (!canvas) return

      function onPointerMove(e) {
        const rect = canvas.getBoundingClientRect()
        const px = e.clientX - rect.left
        const py = e.clientY - rect.top
        // Project each node from layout (0-1000 ortho) to canvas pixel space
        // ortho mapping: layout.x / 1000 → 0..rect.width; layout.y / 1000 → 0..rect.height
        let bestId = null, bestDist = Infinity
        for (const node of nodes) {
          const pos = layout[node.id]
          if (!pos) continue
          const projX = (pos.x / 1000) * rect.width
          const projY = (pos.y / 1000) * rect.height
          const dist = Math.hypot(px - projX, py - projY)
          const radius = computeRadius(node.count, maxCount, 'desktop')
          const pickRadius = radius + 8  // UI-SPEC §Node Geometry pointer pick: +8 device px forgiving margin
          if (dist <= pickRadius && dist < bestDist) {
            bestId = node.id
            bestDist = dist
          }
        }
        if (bestId !== hoveredSkillId) {
          setHoveredSkillId(bestId)
          if (onHoverSkill) onHoverSkill(bestId)
        }
      }

      function onPointerLeave() {
        if (hoveredSkillId !== null) {
          setHoveredSkillId(null)
          if (onHoverSkill) onHoverSkill(null)
        }
      }

      canvas.addEventListener('pointermove', onPointerMove)
      canvas.addEventListener('pointerleave', onPointerLeave)
      return () => {
        canvas.removeEventListener('pointermove', onPointerMove)
        canvas.removeEventListener('pointerleave', onPointerLeave)
      }
    }, [nodes, layout, hoveredSkillId, onHoverSkill])
    ```
