    Theme reactivity useEffect([theme]) per UI-SPEC §"Theme Reactivity" Option A + RESEARCH §8:
    ```
    useEffect(() => {
      if (!materialRef.current) return
      const root = document.documentElement
      const cs = getComputedStyle(root)
      materialRef.current.uniforms.uHaloColor.value = parseCSSColor(cs.getPropertyValue('--color-constellation-halo'))
      materialRef.current.uniforms.uStrokeMix.value = theme === 'light' ? 1.0 : 0.0
      // re-upload strokeColor per-vertex attribute for theme switch
      const strokeAttr = geometryRef.current.getAttribute('strokeColor')
      nodes.forEach((node, i) => {
        const c = parseCSSColor(theme === 'light' ? (LIGHT_THEME_STROKES[node.category] ?? '#00000000') : '#00000000')
        strokeAttr.setXYZ(i, c.r, c.g, c.b)
      })
      strokeAttr.needsUpdate = true
      // edge color uniform update (used by LineBasicMaterial vertexColors — re-upload edge color attribute)
      const edgeColorAttr = edgeGeometryRef.current.getAttribute('color')
      edges.forEach((edge, i) => {
        const c = parseCSSColor(cs.getPropertyValue(edge.weight >= 2 ? '--color-constellation-edge-heavy' : '--color-constellation-edge'))
        edgeColorAttr.setXYZ(i * 2, c.r, c.g, c.b)
        edgeColorAttr.setXYZ(i * 2 + 1, c.r, c.g, c.b)
      })
      edgeColorAttr.needsUpdate = true
      // No explicit renderer.render() call needed — Slice 3 rAF loop picks up next frame.
      // BUT Slice 2 has no rAF — so DO call renderer.render() here to update within one rAF (UI-SPEC §Theme Reactivity contract: "within one rAF frame")
      rendererRef.current.render(sceneRef.current, cameraRef.current)
    }, [theme, nodes, edges])
    ```
