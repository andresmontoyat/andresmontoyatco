    rAF + visibility effect (NEW useEffect, depends [] — runs once on mount; per RESEARCH §10 canonical sketch verbatim):
    ```
    useEffect(() => {
      let rafId
      let lastT = performance.now()

      function tick(t) {
        const dt = (t - lastT) / 1000
        lastT = t
        if (!materialRef.current) return
        // accumulate uTime — drives ambient drift, halo pulse, halo brighten in shader
        materialRef.current.uniforms.uTime.value += dt
        // selected pulse: 1.0 → 1.15 → 1.0 over 2000ms ease-in-out via cos
        const pulsePhase = (materialRef.current.uniforms.uTime.value % 2.0) / 2.0
        materialRef.current.uniforms.uHaloPulse.value = 1.0 + 0.075 * (1.0 - Math.cos(2.0 * Math.PI * pulsePhase))
        // highlight brighten: 0.4 → 0.8 → 0.4 over 3000ms ease-in-out
        const brightPhase = (materialRef.current.uniforms.uTime.value % 3.0) / 3.0
        materialRef.current.uniforms.uHighlightAlpha.value = 0.4 + 0.2 * (1.0 - Math.cos(2.0 * Math.PI * brightPhase))
        rendererRef.current.render(sceneRef.current, cameraRef.current)
        rafId = requestAnimationFrame(tick)
      }

      function onVisibility() {
        if (document.visibilityState === 'hidden') {
          if (rafId !== null) { cancelAnimationFrame(rafId); rafId = null }  // Pitfall 15 sentinel
        } else if (rafId === null) {
          lastT = performance.now()  // Pitfall 16 — reset dt to avoid huge jump
          rafId = requestAnimationFrame(tick)
        }
      }

      rafId = requestAnimationFrame(tick)
      document.addEventListener('visibilitychange', onVisibility)
      return () => {
        if (rafId !== null) cancelAnimationFrame(rafId)
        document.removeEventListener('visibilitychange', onVisibility)
      }
    }, [])
    ```
