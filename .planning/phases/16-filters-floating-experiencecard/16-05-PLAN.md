<!-- ExperienceCard props -->
ExperienceCard({
  selectedNode,        // { id, label, category, color, ... } — the skill whose card is open
  jobs,                // Experience[] — pre-filtered by parent (composeFilters result intersecting selectedSkillId)
  selectedSkills,      // string[] — used for tech-chip active/locked state
  lang,                // 'en' | 'es'
  t,                   // translations[lang]
  onClose(),           // closes dialog (parent maps to onSelectSkill toggle-off)
  onToggleSkill(id),   // tech chip click
})

<!-- Experience shape (from src/data/experience.js) -->
{
  title: { en, es },
  company: 'string',
  date: { en, es },
  location: { en, es },
  bullets: { en: string[], es: string[] },
  tech: string[],          // tech ids, lookup against selectedSkills + selectedNode.id
  period: { start, end },
  featured: boolean,
}

<!-- WAI-ARIA dialog contract (D-16-CARD-A11Y) -->
- role="dialog"
- aria-modal="true"
- aria-labelledby points to the heading id
- Initial focus = heading (NOT close button — W3C recommendation)
- Focus trap on Tab / Shift+Tab cycle
- Esc closes + returns focus to opener (handled via parent's rovingEntryRef)
