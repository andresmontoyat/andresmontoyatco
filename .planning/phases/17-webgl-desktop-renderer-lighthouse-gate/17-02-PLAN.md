## Slice 2 vertex shader (~20 LOC GLSL, per RESEARCH §2 lines 58-79)
```glsl
attribute float size;
attribute vec3 color;
attribute float glow;
attribute float halo;
attribute float dim;
attribute vec3 strokeColor;
varying vec3 vColor;
varying float vGlow;
varying float vHalo;
varying float vDim;
varying vec3 vStrokeColor;
uniform float uDpr;
uniform float uHaloPulse;
void main() {
  vColor = color;
  vGlow = glow;
  vHalo = halo;
  vDim = dim;
  vStrokeColor = strokeColor;
  gl_PointSize = size * uDpr * (1.0 + (uHaloPulse - 1.0) * halo);
  gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
}
```

## Slice 2 fragment shader (~25 LOC GLSL, per RESEARCH §2 lines 80-97)
```glsl
varying vec3 vColor;
varying float vGlow;
varying float vHalo;
varying float vDim;
varying vec3 vStrokeColor;
uniform vec3 uHaloColor;
uniform float uHaloAlpha;
uniform float uStrokeMix;
void main() {
  vec2 uv = gl_PointCoord - vec2(0.5);
  float r = length(uv);
  float core = 1.0 - smoothstep(0.45, 0.5, r);
  float strokeBand = smoothstep(0.40, 0.45, r) - smoothstep(0.45, 0.5, r);
  float haloRing = smoothstep(0.45, 0.5, r) - smoothstep(0.5, 0.7, r);
  vec3 col = mix(vColor, vStrokeColor, strokeBand * uStrokeMix);
  col = mix(col, uHaloColor, vHalo * haloRing);
  float alpha = (core + strokeBand * uStrokeMix + haloRing * vHalo * uHaloAlpha * 0.6 + vGlow * (1.0 - core) * 0.3) * vDim;
  if (alpha < 0.01) discard;
  gl_FragColor = vec4(col, alpha);
}
```

## Theme reactivity useEffect (per UI-SPEC §"Theme Reactivity" Option A + RESEARCH §8 + WARNING 5 RGBA edge re-upload)

useEffect(() => {
  if (!materialRef.current || !geometryRef.current || !edgeGeometryRef.current) return
  const root = document.documentElement
  const cs = getComputedStyle(root)
  const haloRaw = cs.getPropertyValue('--color-constellation-halo')
  materialRef.current.uniforms.uHaloColor.value = parseCSSColor(haloRaw)
  materialRef.current.uniforms.uHaloAlpha.value = parseCSSAlpha(haloRaw)
  materialRef.current.uniforms.uStrokeMix.value = theme === 'light' ? 1.0 : 0.0

  // Per-vertex strokeColor attribute (vec3, itemSize=3) — re-upload on theme switch
  const strokeAttr = geometryRef.current.getAttribute('strokeColor')
  nodes.forEach((node, i) => {
    const raw = theme === 'light' ? (LIGHT_THEME_STROKES[node.category] ?? '#00000000') : '#00000000'
    const c = parseCSSColor(raw)
    strokeAttr.setXYZ(i, c.r, c.g, c.b)
  })
  strokeAttr.needsUpdate = true

  // WARNING 5 — Per-vertex edge color attribute RGBA (vec4, itemSize=4) — alpha channel written separately
  const edgeColorAttr = edgeGeometryRef.current.getAttribute('color')
  edges.forEach((edge, i) => {
    const raw = cs.getPropertyValue(edge.weight >= 2 ? '--color-constellation-edge-heavy' : '--color-constellation-edge')
    const c = parseCSSColor(raw)
    const a = parseCSSAlpha(raw)
    edgeColorAttr.setXYZW(i * 2,     c.r, c.g, c.b, a)
    edgeColorAttr.setXYZW(i * 2 + 1, c.r, c.g, c.b, a)
  })
  edgeColorAttr.needsUpdate = true

  // Slice 2 has no rAF — force one render so theme switch lands within one rAF (UI-SPEC contract)
  rendererRef.current.render(sceneRef.current, cameraRef.current)
}, [theme, nodes, edges])
