    ```jsx
    import React, { createContext, useContext, useState, useCallback, useMemo, useEffect } from 'react'
    import translations from './translations'

    const LanguageContext = createContext({
      lang: 'en',
      setLang: () => {},
      t: translations.en,
    })

    function readInitialLang() {
      if (typeof window === 'undefined') return 'en'
      const saved = window.localStorage.getItem('cam-lang')
      if (saved === 'es' || saved === 'en') return saved
      const nav = (window.navigator &amp;&amp; window.navigator.language) || 'en'
      return nav.toLowerCase().startsWith('es') ? 'es' : 'en'
    }

    export function LanguageProvider({ children }) {
      const [lang, setLangState] = useState(readInitialLang)

      useEffect(() => {
        if (typeof document === 'undefined') return
        document.documentElement.lang = lang
        const meta = translations[lang] &amp;&amp; translations[lang].meta
        if (meta) {
          if (meta.title) document.title = meta.title
          const tag = document.querySelector('meta[name="description"]')
          if (tag &amp;&amp; meta.description) tag.setAttribute('content', meta.description)
        }
      }, [lang])

      const setLang = useCallback((next) => {
        if (next !== 'en' &amp;&amp; next !== 'es') return
        setLangState(next)
        if (typeof window !== 'undefined') {
          window.localStorage.setItem('cam-lang', next)
        }
      }, [])

      const value = useMemo(() => ({
        lang,
        setLang,
        t: translations[lang],
      }), [lang, setLang])

      return (
        <LanguageContext.Provider value={value}>
          {children}
        </LanguageContext.Provider>
      )
    }

    export function useLanguage() {
      return useContext(LanguageContext)
    }
    ```

    Critical implementation notes:

    1. **`useState(readInitialLang)` — NOT `useState(readInitialLang())`.**
       Pass the function reference. React calls it once during the first render synchronously, before paint. This is THE SEO-04 fix.

    2. **`useEffect(..., [lang])` runs on the very first commit AND on every change.**
       This guarantees `<html lang>`, title, and meta description sync from the first paint forward — including on cold load when the saved language is `es`.

    3. **The previous code's `document.documentElement.lang = next` inside `setLang` is REMOVED.**
       The useEffect handles it (else first paint with synchronously-resolved `'es'` would still show `<html lang="en">` until the user clicks).

    4. **`if (meta)` guard:** Defensive. Plan 02-01 adds `meta.title`/`meta.description`, but this guard means a missing key never crashes — it just leaves the head tags untouched.

    5. **Validate `next` in `setLang`:** Reject anything that's not `'en'` or `'es'`. Prevents a stray call from persisting garbage to localStorage.

    6. **SSR safety:** `typeof window === 'undefined'` and `typeof document === 'undefined'` guards stay (per CONVENTIONS.md "Error Handling" note that the codebase already does this).

    7. **No new imports** beyond what's already there (`React`, `createContext`, `useContext`, `useState`, `useCallback`, `useMemo`, `useEffect` — already imported on the existing line 1).

    8. **Public API unchanged:** `LanguageProvider` and `useLanguage` exports keep identical names and shape. No consumer file changes required from this plan.
